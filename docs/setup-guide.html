02

⚙ Configure

Set your API key for any provider you have access to.

  ~/.zshrc
# Pick your provider (set one or more)

export OPENANALYST_AUTH_TOKEN="your-openanalyst-key"     # OpenAnalyst
export ANTHROPIC_API_KEY="sk-ant-..."                   # Claude
export OPENAI_API_KEY="sk-..."                         # GPT / Codex
export OPENROUTER_API_KEY="sk-or-..."                  # OpenRouter
export XAI_API_KEY="xai-..."                          # Grok
export BEDROCK_API_KEY="..."                          # Bedrock
PowerShell
# Pick your provider (set one or more)

$env:OPENANALYST_AUTH_TOKEN = "your-openanalyst-key"     # OpenAnalyst
$env:ANTHROPIC_API_KEY = "sk-ant-..."                   # Claude
$env:OPENAI_API_KEY = "sk-..."                         # GPT / Codex
$env:OPENROUTER_API_KEY = "sk-or-..."                  # OpenRouter
$env:XAI_API_KEY = "xai-..."                          # Grok
$env:BEDROCK_API_KEY = "..."                          # Bedrock
03

▶ Use

  Terminal
$ openanalyst

# Connected to OpenAnalyst API
# Model: openanalyst-beta
openanalystStart interactive session
openanalyst "explain this code"Quick one-shot prompt
openanalyst --model gpt-4o "hello"Use OpenAI
openanalyst --model sonnet "fix bug"Use Claude
openanalyst --model grok "review"Use Grok
openanalyst initSetup project config
/helpShow all commands (inside REPL)
/model gpt-4oSwitch model mid-session
